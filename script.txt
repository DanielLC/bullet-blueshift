subroutine missile(accel, angle, time)
	setParameters("star.png", 0.02, 2)
	accelerate(accel, angle, time)
end

subroutine spiral(accel, initialAngle, deltaAngle, deltaTime, airTime)
	wait(0.1)
	angle = initialAngle
	while true:
		spawn missile(accel, angle, airTime)
		wait(deltaTime)
		angle += deltaAngle
		angle %= 360
	end
end

subroutine enemy()
	#displace(0.3, 0, -0.3)
	setParameters("star.png", 0.2, 2)
	emitter spiral(0.2, 0, 60, 0.03, 4)
	wait(10)
end

spawn enemy()
