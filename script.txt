subroutine missile(accel, angle, time)
	setEnemyBullet(1)
	setParameters("star2.png", 0.02, 2)
	accelerate(accel, angle, time)
end

subroutine missile2(accel, angle, accelTime, coastTime)
	setAllyBullet(1)
	setParameters("star2.png", 0.02, 2)
	accelerate(accel, angle, accelTime)
	accelerate(0, 0, coastTime)
end

subroutine spiral(accel, initialAngle, deltaAngle, deltaTime, airTime)
	var angle = initialAngle
	while true:
		spawn missile(accel, angle, airTime)
		wait(deltaTime)
		angle += deltaAngle
		angle %= 360
	end
end

subroutine zigzagSpiral(accel, branches, minAngle, maxAngle, deltaAngle, deltaAngle2, deltaTime, airTime)
	var i = 0
	var angle = minAngle
	while true:
		while angle < maxAngle:
			i = 0
			while i < branches:
				spawn missile(accel, angle + i*360/branches, airTime)
				i += 1
			end
			wait(deltaTime)
			angle += deltaAngle
		end
		while angle > minAngle:
			i = 0
			while i < branches:
				spawn missile(accel, angle + i*360/branches, airTime)
				i += 1
			end
			wait(deltaTime)
			angle -= deltaAngle2
		end
	end
end

#It's easier just to use an emitter to shoot. Then I don't have to tie the fire rate to the framerate.
#I think the problem here is once it gets to the player's current position in spacetime, it assumes they're dead.
subroutine shoot()
	var a = 1
	var accelTime = 0.2
	var coastTime = 5
	var fireDelay = 0.15
	var accuracy = 10
	
	while true:
		#print("Pew!")
		var dir = mouseDirection()
		#print(dir)
		if !isNaN(dir):
			spawn missile2(a, dir + accuracy * (random() - 0.5), accelTime, coastTime)
		end
		wait(fireDelay)
	end
end

subroutine exploder(size, brightness)
	setParameters("star2.png", 0.03, 3)
	accelerate(0.2, 0, 1)
	explode(size, brightness)
end

subroutine dot(a, accelTime, restTime)
	setParameters("star2.png", 0.03, 3)
	accelerate(a, 180, accelTime)
	accelerate(0, 0, restTime)
end

subroutine dottedLine(a, dist, n)
	var x = dist/(2.0*n)
	var t = sqrt(pow(x + 1.0/a, 2) - 1.0/(a*a))
	var tau = asinh(t*a)
	var v = tanh(a*tau)
	var gamma = 1.0/sqrt(1-v*v)
	var restTime = 2.0*x/v
	var restProperTime = restTime/gamma
	print([a, dist, n, x, t, tau, v, gamma, restTime, restProperTime])

	setParameters("star2.png", 0.03, 0)
	accelerate(a, 0, tau)
	while n > 0:
		print("Spawning dot")
		spawn dot(a, tau, n * restTime)
		accelerate(0, 0, restProperTime)
		n -= 1
	end
	call dot(a, tau, 0)
end

subroutine move(a, dist, dir)
	var t = sqrt(pow(dist*0.5 + 1.0/a, 2) - 1.0/(a*a))
	var tau = asinh(t*a)
	accelerate(a, dir, tau)
	accelerate(a, dir+180, tau)
end

subroutine explodingStarBullet(a, dist, n)
	#Calling this seems to override the perameters of the function you're in. This is a big bug.
	call move(a, dist, 0)
	#Rotate didn't work
	rotate(36)
	call dottedLine(a, dist * 1.902113032590307, n)
end
